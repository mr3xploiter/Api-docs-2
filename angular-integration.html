Angular Integration

Complete guide to integrating the Media Upload API with Angular applications (Angular 14+).

Installation

npm install axios

Service Implementation

// media-upload.service.ts
import { Injectable } from '@angular/core';
import { HttpClient, HttpEvent, HttpProgressEvent } from '@angular/common/http';
import { Observable } from 'rxjs';

export interface UploadResponse {
  success: boolean;
  type: string;
  filename: string;
  url: string;
  size: string;
  uploadDate: string;
}

@Injectable({
  providedIn: 'root'
})
export class MediaUploadService {
  private apiUrl = 'http://localhost:5000/api';

  constructor(private http: HttpClient) {}

  uploadImage(file: File): Observable<UploadResponse> {
    const formData = new FormData();
    formData.append('file', file);

    return this.http.post<UploadResponse>(
      `${this.apiUrl}/upload/image`,
      formData
    );
  }

  uploadVideo(file: File): Observable<UploadResponse> {
    const formData = new FormData();
    formData.append('file', file);

    return this.http.post<UploadResponse>(
      `${this.apiUrl}/upload/video`,
      formData
    );
  }

  uploadWithProgress(
    file: File,
    type: 'image' | 'video' = 'image'
  ): Observable<HttpEvent<UploadResponse>> {
    const formData = new FormData();
    formData.append('file', file);

    return this.http.post<UploadResponse>(
      `${this.apiUrl}/upload/${type}`,
      formData,
      { reportProgress: true, responseType: 'json' }
    );
  }

  getMediaInfo(filename: string): Observable<any> {
    return this.http.get(`${this.apiUrl}/media/${filename}`);
  }

  deleteMedia(filename: string): Observable<any> {
    return this.http.delete(`${this.apiUrl}/media/${filename}`);
  }

  listMedia(type: 'images' | 'videos'): Observable<any> {
    return this.http.get(`${this.apiUrl}/media/list/${type}`);
  }

  getStatistics(): Observable<any> {
    return this.http.get(`${this.apiUrl}/media/statistics`);
  }
}

Component Implementation

// uploader.component.ts
import { Component } from '@angular/core';
import { MediaUploadService, UploadResponse } from './services/media-upload.service';

@Component({
  selector: 'app-uploader',
  templateUrl: './uploader.component.html',
  styleUrls: ['./uploader.component.css']
})
export class UploaderComponent {
  file: File | null = null;
  uploading = false;
  error = '';
  url = '';
  progress = 0;
  uploadedFiles: UploadResponse[] = [];

  constructor(private mediaUploadService: MediaUploadService) {}

  handleFileSelect(event: Event): void {
    const input = event.target as HTMLInputElement;
    this.file = input.files ? input.files[0] : null;
    this.error = '';
  }

  async uploadFile(): Promise<void> {
    if (!this.file) return;

    this.uploading = true;
    this.error = '';

    try {
      const response = await this.mediaUploadService
        .uploadImage(this.file)
        .toPromise();

      if (response?.success) {
        this.url = response.url;
        this.uploadedFiles.push(response);
        this.file = null;
      } else {
        this.error = 'Upload failed';
      }
    } catch (err) {
      this.error = 'Upload error: ' + (err instanceof Error ? err.message : 'Unknown error');
    } finally {
      this.uploading = false;
    }
  }

  uploadWithProgress(): void {
    if (!this.file) return;

    this.uploading = true;
    this.error = '';

    this.mediaUploadService
      .uploadWithProgress(this.file)
      .subscribe({
        next: (event: any) => {
          if (event.type === 4) {
            this.url = event.body.url;
            this.uploadedFiles.push(event.body);
            this.file = null;
          }
        },
        error: (err) => {
          this.error = 'Upload failed: ' + err.message;
          this.uploading = false;
        },
        complete: () => {
          this.uploading = false;
          this.progress = 0;
        }
      });
  }

  deleteFile(filename: string): void {
    this.mediaUploadService
      .deleteMedia(filename)
      .subscribe({
        next: () => {
          this.uploadedFiles = this.uploadedFiles.filter(f => f.filename !== filename);
        },
        error: (err) => {
          console.error('Delete failed:', err);
        }
      });
  }
}

Template

<!-- uploader.component.html -->
<div class="uploader-container">
  <h2>Upload Media</h2>

  <div class="upload-form">
    <input 
      type="file" 
      (change)="handleFileSelect($event)"
      #fileInput
    />
    <button 
      (click)="uploadFile()"
      [disabled]="!file || uploading"
      class="btn-primary"
    >
      {{ uploading ? 'Uploading...' : 'Upload' }}
    </button>
  </div>

  <p *ngIf="error" class="error">{{ error }}</p>

  <div *ngIf="url" class="preview">
    <img [src]="url" alt="Uploaded" />
  </div>

  <div *ngIf="uploading" class="progress">
    <div [style.width.%]="progress"></div>
  </div>

  <div class="uploaded-list">
    <h3>Uploaded Files</h3>
    <div *ngFor="let file of uploadedFiles" class="file-item">
      <img [src]="file.url" [alt]="file.filename" />
      <p>{{ file.filename }}</p>
      <button (click)="deleteFile(file.filename)" class="btn-danger">
        Delete
      </button>
    </div>
  </div>
</div>

Drag and Drop Upload

// uploader.component.ts
import { Component, HostListener } from '@angular/core';

@Component({
  selector: 'app-uploader',
  template: `
    <div 
      class="drop-zone"
      (drop)="onDrop($event)"
      (dragover)="onDragOver($event)"
      (dragleave)="onDragLeave($event)"
      [class.active]="dragover"
    >
      <p>Drop files here</p>
      <input 
        type="file" 
        (change)="handleFileSelect($event)"
        #fileInput
        hidden
      />
      <button (click)="fileInput.click()">
        Choose File
      </button>
    </div>
  `
})
export class UploaderComponent {
  dragover = false;

  onDragOver(event: DragEvent): void {
    event.preventDefault();
    this.dragover = true;
  }

  onDragLeave(event: DragEvent): void {
    event.preventDefault();
    this.dragover = false;
  }

  onDrop(event: DragEvent): void {
    event.preventDefault();
    this.dragover = false;
    const files = event.dataTransfer?.files;
    if (files) {
      this.file = files[0];
    }
  }
}
