Best Practices

Industry best practices and optimization techniques for using the Media Upload API effectively and securely.

Client-Side Best Practices

1. Validate Files Before Upload

// Always validate on client-side first
const validateFile = (file) => {
  // Check file size
  const MAX_SIZE = 10 * 1024 * 1024; // 10MB
  if (file.size > MAX_SIZE) {
    return { valid: false, error: 'File too large' };
  }

  // Check file type
  const ALLOWED_TYPES = [
    'image/jpeg',
    'image/png',
    'image/webp',
    'image/gif'
  ];

  if (!ALLOWED_TYPES.includes(file.type)) {
    return { valid: false, error: 'Invalid file type' };
  }

  // Check file extension
  const extension = file.name.split('.').pop().toLowerCase();
  const ALLOWED_EXTENSIONS = ['jpg', 'jpeg', 'png', 'webp', 'gif'];

  if (!ALLOWED_EXTENSIONS.includes(extension)) {
    return { valid: false, error: 'Invalid file extension' };
  }

  return { valid: true };
};

// Usage
const file = fileInput.files[0];
const validation = validateFile(file);

if (!validation.valid) {
  alert('Error: ' + validation.error);
  return;
}

2. Handle Errors Gracefully

// Implement comprehensive error handling
const uploadFile = async (file) => {
  try {
    // Validate before upload
    const validation = validateFile(file);
    if (!validation.valid) {
      throw new Error(validation.error);
    }

    // Perform upload
    const response = await fetch('http://localhost:5000/api/upload/image', {
      method: 'POST',
      body: formData
    });

    // Check HTTP status
    if (!response.ok) {
      if (response.status === 413) {
        throw new Error('File too large');
      } else if (response.status === 429) {
        throw new Error('Too many uploads. Please wait before trying again.');
      } else {
        throw new Error(`HTTP Error: ${response.status}`);
      }
    }

    const data = await response.json();

    // Check API response
    if (!data.success) {
      throw new Error(data.error || 'Upload failed');
    }

    return data;

  } catch (error) {
    // Log error with context
    console.error('Upload error:', {
      message: error.message,
      file: file.name,
      size: file.size,
      timestamp: new Date().toISOString()
    });

    // Show user-friendly message
    showUserMessage(
      'Upload failed: ' + error.message,
      'error'
    );

    throw error;
  }
};

3. Implement Retry Logic

// Retry with exponential backoff
const uploadWithRetry = async (file, maxRetries = 3) => {
  let lastError;

  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await uploadFile(file);
    } catch (error) {
      lastError = error;

      // Don't retry on validation errors
      if (error.message.includes('Invalid')) {
        throw error;
      }

      // Wait before retrying (exponential backoff)
      if (attempt < maxRetries) {
        const delay = Math.pow(2, attempt - 1) * 1000; // 1s, 2s, 4s
        console.log(`Retry attempt ${attempt} after ${delay}ms`);
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
  }

  throw new Error(`Upload failed after ${maxRetries} attempts: ${lastError.message}`);
};

4. Show Upload Progress

Always provide visual feedback for uploads, especially for larger files:

// User-friendly progress tracking
const uploadWithProgressUI = async (file) => {
  const progressContainer = document.getElementById('progress');
  const progressBar = document.getElementById('progressBar');
  const progressText = document.getElementById('progressText');

  progressContainer.style.display = 'block';

  try {
    await uploadWithProgress(file, (progress) => {
      // Update visual progress bar
      progressBar.style.width = progress + '%';
      progressText.textContent = Math.round(progress) + '%';

      // Update estimatedTime
      if (progress > 0 && progress < 100) {
        const elapsed = Date.now() - startTime;
        const estimated = (elapsed / progress) * (100 - progress);
        console.log(`~${Math.round(estimated / 1000)}s remaining`);
      }
    });

    // Show success state
    progressBar.style.width = '100%';
    progressText.textContent = 'Complete!';
    setTimeout(() => {
      progressContainer.style.display = 'none';
    }, 2000);

  } catch (error) {
    progressContainer.style.display = 'none';
    throw error;
  }
};

File Type Validation

// Deep validation of file type
const validateFileType = (file, expectedType) => {
  // Check MIME type
  if (!file.type.startsWith(expectedType)) {
    return false;
  }

  // Check magic numbers for image files
  if (expectedType === 'image') {
    const magicNumbers = {
      'FFD8FF': 'image/jpeg',     // JPEG
      '89504E47': 'image/png',    // PNG
      '47494638': 'image/gif',    // GIF
      '52494646': 'image/webp'    // WebP
    };

    // Read first few bytes
    const reader = new FileReader();
    reader.onload = (e) => {
      const arr = new Uint8Array(e.target.result).subarray(0, 4);
      let header = '';
      for (let i = 0; i < arr.length; i++) {
        header += arr[i].toString(16);
      }
      return magicNumbers[header];
    };
    reader.readAsArrayBuffer(file);
  }

  return true;
};

Performance Optimization

1. Image Compression

// Compress image before upload (optional)
const compressImage = async (file, quality = 0.8) => {
  return new Promise((resolve) => {
    const reader = new FileReader();

    reader.onload = (e) => {
      const img = new Image();
      img.onload = () => {
        const canvas = document.createElement('canvas');
        canvas.width = img.width;
        canvas.height = img.height;

        const ctx = canvas.getContext('2d');
        ctx.drawImage(img, 0, 0);

        canvas.toBlob(
          (blob) => {
            resolve(blob);
          },
          'image/jpeg',
          quality
        );
      };
      img.src = e.target.result;
    };

    reader.readAsDataURL(file);
  });
};

// Usage: Reduce file size before uploading
const originalFile = fileInput.files[0];
const compressedFile = await compressImage(originalFile, 0.8);
const result = await uploadFile(compressedFile);

2. Chunked Upload for Large Files

// Split large files into chunks
const CHUNK_SIZE = 5 * 1024 * 1024; // 5MB chunks

const uploadInChunks = async (file) => {
  const totalChunks = Math.ceil(file.size / CHUNK_SIZE);

  for (let i = 0; i < totalChunks; i++) {
    const start = i * CHUNK_SIZE;
    const end = Math.min(start + CHUNK_SIZE, file.size);
    const chunk = file.slice(start, end);

    const formData = new FormData();
    formData.append('file', chunk);
    formData.append('chunkIndex', i);
    formData.append('totalChunks', totalChunks);

    await fetch('http://localhost:5000/api/upload/chunk', {
      method: 'POST',
      body: formData
    });

    // Update progress
    const progress = ((i + 1) / totalChunks) * 100;
    onProgress?.(progress);
  }
};

3. Caching Strategy

// Cache file metadata to reduce API calls
const fileMetadataCache = new Map();

const getFileInfoCached = async (filename, ttl = 3600000) => {
  // Check cache first
  if (fileMetadataCache.has(filename)) {
    const cached = fileMetadataCache.get(filename);
    if (Date.now() - cached.timestamp < ttl) {
      return cached.data;
    }
  }

  // Fetch from server if not cached
  const info = await getFileInfo(filename);

  // Cache the result
  fileMetadataCache.set(filename, {
    data: info,
    timestamp: Date.now()
  });

  return info;
};
