Svelte Component

<script>
  let file;
  let uploading = false;
  let error = '';
  let url = '';
  let uploadedFiles = [];

  const handleFileSelect = (e) => {
    file = e.target.files[0];
    error = '';
  };

  const uploadFile = async () => {
    if (!file) return;

    const formData = new FormData();
    formData.append('file', file);

    uploading = true;
    error = '';

    try {
      const response = await fetch(
        'http://localhost:5000/api/upload/image',
        {
          method: 'POST',
          body: formData
        }
      );

      const data = await response.json();

      if (data.success) {
        url = data.url;
        uploadedFiles = [...uploadedFiles, data];
        file = null;
      } else {
        error = data.error;
      }
    } catch (err) {
      error = 'Upload failed: ' + err.message;
    } finally {
      uploading = false;
    }
  };

  const deleteFile = async (filename) => {
    try {
      const response = await fetch(
        `http://localhost:5000/api/media/${filename}`,
        { method: 'DELETE' }
      );

      if (response.ok) {
        uploadedFiles = uploadedFiles.filter(f => f.filename !== filename);
      }
    } catch (err) {
      console.error('Delete failed:', err);
    }
  };
</script>

<div class="uploader">
  <input 
    type="file" 
    on:change={handleFileSelect}
  />
  <button 
    on:click={uploadFile}
    disabled={!file || uploading}
  >
    {uploading ? 'Uploading...' : 'Upload'}
  </button>

  {#if error}
    <p class="error">{error}</p>
  {/if}

  {#if url}
    <img src={url} alt="Uploaded" />
  {/if}

  <div class="uploaded-list">
    {#each uploadedFiles as item (item.filename)}
      <div class="file-item">
        <img src={item.url} alt={item.filename} />
        <p>{item.filename}</p>
        <button on:click={() => deleteFile(item.filename)}>
          Delete
        </button>
      </div>
    {/each}
  </div>
</div>

Custom Store

// stores/mediaStore.js
import { writable } from 'svelte/store';

export const createMediaStore = () => {
  const { subscribe, set, update } = writable({
    files: [],
    uploading: false,
    error: null
  });

  return {
    subscribe,
    uploadFile: async (file) => {
      update(state => ({ ...state, uploading: true, error: null }));

      const formData = new FormData();
      formData.append('file', file);

      try {
        const response = await fetch(
          'http://localhost:5000/api/upload/image',
          { method: 'POST', body: formData }
        );

        const data = await response.json();

        if (data.success) {
          update(state => ({
            ...state,
            files: [...state.files, data],
            uploading: false
          }));
        } else {
          update(state => ({
            ...state,
            error: data.error,
            uploading: false
          }));
        }
      } catch (err) {
        update(state => ({
          ...state,
          error: err.message,
          uploading: false
        }));
      }
    },
    deleteFile: async (filename) => {
      try {
        await fetch(
          `http://localhost:5000/api/media/${filename}`,
          { method: 'DELETE' }
        );

        update(state => ({
          ...state,
          files: state.files.filter(f => f.filename !== filename)
        }));
      } catch (err) {
        console.error('Delete failed:', err);
      }
    }
  };
};

export const mediaStore = createMediaStore();

Using Store in Component

<script>
  import { mediaStore } from './stores/mediaStore.js';

  let file;

  const handleFileSelect = (e) => {
    file = e.target.files[0];
  };

  const handleUpload = () => {
    if (file) {
      mediaStore.uploadFile(file);
      file = null;
    }
  };
</script>

<div>
  <input type="file" on:change={handleFileSelect} />
  <button on:click={handleUpload} disabled={$mediaStore.uploading}>
    {$mediaStore.uploading ? 'Uploading...' : 'Upload'}
  </button>

  {#if $mediaStore.error}
    <p class="error">{$mediaStore.error}</p>
  {/if}

  <div>
    {#each $mediaStore.files as file (file.filename)}
      <div>
        <img src={file.url} alt={file.filename} />
        <button on:click={() => mediaStore.deleteFile(file.filename)}>
          Delete
        </button>
      </div>
    {/each}
  </div>
</div>

SvelteKit Server Load

// +page.server.js
export async function POST({ request }) {
  const formData = await request.formData();
  const file = formData.get('file');

  try {
    const uploadFormData = new FormData();
    uploadFormData.append('file', file);

    const response = await fetch(
      'http://localhost:5000/api/upload/image',
      {
        method: 'POST',
        body: uploadFormData
      }
    );

    const data = await response.json();

    if (data.success) {
      return { success: true, url: data.url };
    } else {
      return { success: false, error: data.error };
    }
  } catch (error) {
    return { success: false, error: error.message };
  }
}

Drag and Drop Support

<script>
  let dragover = false;
  let file;

  const handleDrop = (e) => {
    e.preventDefault();
    dragover = false;
    file = e.dataTransfer.files[0];
  };

  const handleDragOver = (e) => {
    e.preventDefault();
    dragover = true;
  };
</script>

<div 
  class="drop-zone"
  on:drop={handleDrop}
  on:dragover={handleDragOver}
  on:dragleave={() => dragover = false}
  class:active={dragover}
>
  <p>Drop files here</p>
  <input 
    type="file" 
    bind:files
    hidden
    on:change={(e) => file = e.target.files[0]}
  />
  <button on:click={() => document.querySelector('input').click()}>
    Choose File
  </button>
</div>
