Vue.js Integration

Complete guide to integrating the Media Upload API with Vue.js applications (Vue 2 & 3).

Installation

npm install axios

Vue 3 Composition API

<template>
  <div class="uploader">
    <input 
      type="file" 
      @change="handleFileSelect"
      accept="image/*,video/*"
    />
    <button 
      @click="uploadFile"
      :disabled="!file || uploading"
    >
      {{ uploading ? 'Uploading...' : 'Upload' }}
    </button>
    <p v-if="error" class="error">{{ error }}</p>
    <img v-if="url" :src="url" :alt="filename" />
  </div>
</template>

<script setup>
import { ref } from 'vue';
import axios from 'axios';

const file = ref(null);
const uploading = ref(false);
const error = ref('');
const url = ref('');
const filename = ref('');

const handleFileSelect = (e) => {
  file.value = e.target.files[0];
  error.value = '';
};

const uploadFile = async () => {
  if (!file.value) return;

  const formData = new FormData();
  formData.append('file', file.value);

  uploading.value = true;
  error.value = '';

  try {
    const response = await axios.post(
      'http://localhost:5000/api/upload/image',
      formData
    );

    if (response.data.success) {
      url.value = response.data.url;
      filename.value = response.data.originalName;
      file.value = null;
    } else {
      error.value = response.data.error;
    }
  } catch (err) {
    error.value = 'Upload failed: ' + err.message;
  } finally {
    uploading.value = false;
  }
};
</script>

Custom Composable

// composables/useMediaUpload.js
import { ref } from 'vue';
import axios from 'axios';

export function useMediaUpload(apiUrl = 'http://localhost:5000') {
  const uploading = ref(false);
  const error = ref('');
  const progress = ref(0);

  const upload = async (file, type = 'image') => {
    const formData = new FormData();
    formData.append('file', file);

    uploading.value = true;
    error.value = '';

    try {
      const response = await axios.post(
        `${apiUrl}/api/upload/${type}`,
        formData,
        {
          onUploadProgress: (e) => {
            progress.value = (e.loaded / e.total) * 100;
          }
        }
      );

      if (response.data.success) {
        return response.data;
      } else {
        throw new Error(response.data.error);
      }
    } catch (err) {
      error.value = err.message;
      throw err;
    } finally {
      uploading.value = false;
      progress.value = 0;
    }
  };

  return {
    upload,
    uploading,
    error,
    progress
  };
}

Drag and Drop Upload

<template>
  <div 
    class="drop-zone"
    @dragover="dragover = true"
    @dragleave="dragover = false"
    @drop="handleDrop"
    :class="{ active: dragover }"
  >
    <p v-if="!preview">Drop files here</p>
    <img v-else :src="preview" />
    <input 
      type="file" 
      @change="handleFileSelect"
      hidden
      ref="fileInput"
    />
    <button @click="$refs.fileInput.click()">
      Choose File
    </button>
  </div>
</template>

<script setup>
import { ref } from 'vue';

const file = ref(null);
const preview = ref('');
const dragover = ref(false);

const handleFileSelect = (e) => {
  file.value = e.target.files[0];
  createPreview();
};

const handleDrop = (e) => {
  e.preventDefault();
  dragover.value = false;
  file.value = e.dataTransfer.files[0];
  createPreview();
};

const createPreview = () => {
  if (file.value) {
    const reader = new FileReader();
    reader.onload = (e) => {
      preview.value = e.target.result;
    };
    reader.readAsDataURL(file.value);
  }
};
</script>

Error Handling

<script setup>
import { ref } from 'vue';

const uploadFile = async (file) => {
  const formData = new FormData();
  formData.append('file', file);

  try {
    const response = await fetch(
      'http://localhost:5000/api/upload/image',
      {
        method: 'POST',
        body: formData
      }
    );

    const data = await response.json();

    if (!data.success) {
      switch (data.code) {
        case 'FILE_TOO_LARGE':
          error.value = 'File is too large. Max 10MB.';
          break;
        case 'INVALID_FILE_TYPE':
          error.value = 'Invalid file type. Use JPG, PNG, WebP, or GIF.';
          break;
        default:
          error.value = data.error;
      }
      return;
    }

    return data.url;
  } catch (err) {
    error.value = 'Upload failed: ' + err.message;
  }
};
</script>
